##### 2. 积分奖励规则引擎

**积分计算服务**：
```javascript
class PointsService {
  // 计算课时完成积分
  static calculateLessonPoints(lesson, userLevel, completionData) {
    let basePoints = lesson.pointsReward?.basePoints || 10;
    let totalPoints = basePoints;
    
    // 等级加成
    const levelMultiplier = lesson.pointsReward?.levelMultiplier || 1.0;
    totalPoints *= (1 + (userLevel - 1) * 0.1) * levelMultiplier;
    
    // 奖励条件加成
    if (lesson.pointsReward?.bonusConditions) {
      lesson.pointsReward.bonusConditions.forEach(bonus => {
        if (this.checkBonusCondition(bonus.condition, completionData)) {
          totalPoints *= bonus.multiplier;
        }
      });
    }
    
    return Math.floor(totalPoints);
  }
  
  // 发放积分并记录
  static async awardPoints(userId, pointsData) {
    const session = await mongoose.startSession();
    session.startTransaction();
    
    try {
      // 创建积分记录
      const pointsRecord = new PointsRecord({
        user: userId,
        type: pointsData.type,
        source: pointsData.source,
        points: pointsData.points,
        description: pointsData.description,
        metadata: pointsData.metadata
      });
      
      await pointsRecord.save({ session });
      
      // 更新用户总积分
      await User.findByIdAndUpdate(
        userId,
        { $inc: { 'learningStats.totalPoints': pointsData.points } },
        { session }
      );
      
      await session.commitTransaction();
      return pointsRecord;
    } catch (error) {
      await session.abortTransaction();
      throw error;
    } finally {
      session.endSession();
    }
  }
}
```

    const user = await User.findById(userId);
    const lesson = await Course.findOne(
      { 'lessons._id': lessonId },

### 3.1 关卡解锁系统

#### 解锁逻辑实现
```javascript
// courseController.js - 检查课时解锁状态
const checkUnlockStatus = async (req, res) => {
  try {
    const { courseId, lessonId } = req.params;
    const userId = req.user._id;
    
    const enrollment = await Enrollment.findOne({
      student: userId,
      course: courseId
    }).populate('course');
    
    if (!enrollment) {
      return res.status(404).json({ success: false, message: '未报名此课程' });
    }
    
    const lesson = enrollment.course.lessons.id(lessonId);
    if (!lesson) {
      return res.status(404).json({ success: false, message: '课时不存在' });
    }
    
    // 检查解锁条件
    const isUnlocked = await checkLessonUnlockConditions(enrollment, lesson);
    
    res.json({
      success: true,
      data: {
        isUnlocked,
        unlockConditions: lesson.unlockConditions,
        progress: enrollment.progress
      }
    });
  } catch (error) {
    res.status(500).json({ success: false, message: '检查解锁状态失败' });
  }
};

// 解锁条件检查函数
const checkLessonUnlockConditions = async (enrollment, lesson) => {
  // 如果没有解锁条件，默认解锁
  if (!lesson.unlockConditions || lesson.unlockConditions.length === 0) {
    return true;
  }
  
  for (const condition of lesson.unlockConditions) {
    switch (condition.condition) {
      case 'complete':
        const isCompleted = enrollment.progress.completedLessons.some(
          cl => cl.lessonId.toString() === condition.type
        );
        if (!isCompleted) return false;
        break;
        
      case 'score_above':
        // 检查分数条件
        const score = getLessonScore(enrollment, condition.type);
        if (score < condition.value) return false;
        break;
        
      case 'time_spent':
        // 检查学习时长条件
        const timeSpent = getLessonTimeSpent(enrollment, condition.type);
        if (timeSpent < condition.value) return false;
        break;
    }
  }
  
  return true;
};
```

### 3.2 经验值系统

#### 经验值计算与奖励
```javascript
// 完成课时获得经验值
const completeLesson = async (req, res) => {
  try {
    const { courseId, lessonId } = req.params;
    const { timeSpent = 0, score = 0 } = req.body;
    const userId = req.user._id;
    
    const enrollment = await Enrollment.findOne({
      student: userId,
      course: courseId
    }).populate('course');
    
    const lesson = enrollment.course.lessons.id(lessonId);
    
    // 计算经验值奖励
    let xpEarned = lesson.xpReward || 10;
    
    // 根据表现给予额外奖励
    if (score >= 90) xpEarned += 5; // 优秀表现奖励
    if (timeSpent <= lesson.duration * 0.8) xpEarned += 3; // 高效学习奖励
    
    // 更新学习进度
    await enrollment.updateProgress(lessonId, timeSpent);
    
    // 更新经验值
    enrollment.progress.earnedXP += xpEarned;
    
    // 更新用户总经验值
    await User.findByIdAndUpdate(userId, {
      $inc: { 'learningStats.totalPoints': xpEarned }
    });
    
    // 检查成就
    const achievements = await checkAchievements(enrollment, lesson, { score, timeSpent });
    
    await enrollment.save();
    
    res.json({
      success: true,
      data: {
        xpEarned,
        totalXP: enrollment.progress.earnedXP,
        achievements,
        nextLessonUnlocked: await getNextUnlockedLesson(enrollment)
      }
    });
  } catch (error) {
    res.status(500).json({ success: false, message: '完成课时失败' });
  }
};
```

    case 'complete':
      return '完成前置课程后解锁'
    case 'score_above':
      return `需要达到${condition.value}分以上`
    case 'time_spent':
      return `需要学习${condition.value}分钟以上`
    default:
      return '满足条件后解锁'
  }
}

const startLesson = (lesson) => {
  if (lesson.status !== 'locked') {
    emit('lesson-start', lesson)
  }
}
</script>

**影响程度**：中等

- 实现乐观锁机制
